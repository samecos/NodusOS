- [ ] **Step 3: 最小修改——将邻接表与已访问集合改为 `symbol_id` 键**

将 `src/ui/terminal-renderer.ts` 中 `renderCallGraph` 替换为如下实现（保留 ANSI 颜色辅助函数）：

```typescript
  private renderCallGraph(graph: CallGraph): string {
    if (graph.nodes.length === 0) return c('调用图为空', DIM);

    const nodeMap = new Map<string, CallGraphNode>(
      graph.nodes.map(n => [n.symbol_id, n]),
    );

    // 邻接表：symbol_id -> 邻接 symbol_id 列表
    const adjacency = new Map<string, string[]>();
    for (const edge of graph.edges) {
      if (!adjacency.has(edge.from)) adjacency.set(edge.from, []);
      adjacency.get(edge.from)!.push(edge.to);
    }

    const rootId = graph.root_symbol_id;
    const root = nodeMap.get(rootId);
    if (!root) return c('调用图根节点不存在', DIM);

    let out = c('\n调用图', BOLD);
    if (graph.direction === 'callers') {
      out += c(' （上游调用方）', DIM);
    } else if (graph.direction === 'callees') {
      out += c(' （下游被调用方）', DIM);
    } else {
      out += c(' （双向）', DIM);
    }
    out += `\n${c('最大深度:', DIM)} ${c(String(graph.max_depth), YELLOW)}\n`;

    const visited = new Set<string>();

    const formatNode = (id: string): string => {
      const n = nodeMap.get(id);
      if (!n) return c(`[unknown:${id}]`, DIM);
      const fileName = n.file_path.split('/').pop() ?? n.file_path;
      const risk = n.has_risk ? c(' ⚠', YELLOW) : '';
      return `${c(n.symbol_name, BOLD)} ${c(`[${fileName}:${n.line}]`, DIM)}${risk}`;
    };

    const printTree = (id: string, prefix: string, isLast: boolean): string => {
      if (visited.has(id)) {
        return `${prefix}${isLast ? '└─ ' : '├─ '}${formatNode(id)} ${c('(cycle)', RED)}\n`;
      }
      visited.add(id);

      let result = `${prefix}${isLast ? '└─ ' : '├─ '}${formatNode(id)}\n`;
      const kids = adjacency.get(id) ?? [];
      for (let i = 0; i < kids.length; i++) {
        const childLast = i === kids.length - 1;
        const nextPrefix = prefix + (isLast ? '   ' : '│  ');
        result += printTree(kids[i]!, nextPrefix, childLast);
      }
      return result;
    };

    out += printTree(rootId, '  ', true);
    return out;
  }
```

- [ ] **Step 3: 实现方向感知展开**

修改 `src/ui/terminal-renderer.ts` 中 `renderCallGraph`：

1. 在 `nodeMap` 之后计算两个邻接表：
   - `forwardAdj`：`edge.from -> edge.to`（callees 使用）。
   - `reverseAdj`：`edge.to -> edge.from`（callers 使用）。
2. 提取 `formatNode(id)` 与 `printTree(adj, id, ...)` 辅助方法，使 `printTree` 接受邻接表参数。
3. 根据 `graph.direction` 渲染：
   - `'callees'`：根节点 + `forwardAdj`。
   - `'callers'`：根节点 + `reverseAdj`，标题区显示 `上游调用方`。
   - `'both'`：先以 `reverseAdj` 打印 `上游` 子树，再以 `forwardAdj` 打印 `下游` 子树；两棵子树独立 `visited` 集合，避免互相影响深度。

伪代码核心：

```typescript
const forwardAdj = new Map<string, string[]>();
const reverseAdj = new Map<string, string[]>();
for (const edge of graph.edges) {
  if (!forwardAdj.has(edge.from)) forwardAdj.set(edge.from, []);
  forwardAdj.get(edge.from)!.push(edge.to);
  if (!reverseAdj.has(edge.to)) reverseAdj.set(edge.to, []);
  reverseAdj.get(edge.to)!.push(edge.from);
}

const printTree = (adj: Map<string, string[]>, id: string, prefix: string, isLast: boolean, visited: Set<string>): string => { ... };

if (graph.direction === 'callers') {
  out += printTree(reverseAdj, rootId, '  ', true, new Set());
} else if (graph.direction === 'callees') {
  out += printTree(forwardAdj, rootId, '  ', true, new Set());
} else {
  out += `\n${c('上游', BOLD)}\n`;
  out += printTree(reverseAdj, rootId, '  ', true, new Set());
  out += `\n${c('下游', BOLD)}\n`;
  out += printTree(forwardAdj, rootId, '  ', true, new Set());
}
```
