**Goal:** 让 `JsonConfigManager` 能够可靠地监听 `~/.nodus/config.json` 的变更（包括文件创建、修改、删除），去抖动后自动 reload，并把 `ConfigManager` 实例注入 `NodusShell`，使运行时的配置变更即时生效。

**Tech Stack:** TypeScript, ESM, Node.js `fs.watch`, Vitest。

- [ ] **Step 3: 实现 `fs.watch` 监听与去抖动 reload**

将 `src/common/config.ts` 中相关部分替换为以下实现：

```typescript
import { readFileSync, writeFileSync, existsSync, mkdirSync, watch } from 'node:fs';
import type { FSWatcher } from 'node:fs';
import { dirname, basename } from 'node:path';

// ...（其余 import 与类型不变）

export class JsonConfigManager implements ConfigManager {
  private config: NodusConfig;
  private configPath: string;
  private listeners = new Set<(config: NodusConfig) => void>();
  private watcher?: FSWatcher;
  private debounceTimer?: ReturnType<typeof setTimeout>;
  private readonly debounceMs = 100;

  constructor(configPath = join(homedir(), '.nodus', 'config.json')) {
    this.configPath = configPath;
    this.config = this.load();
    this.watchConfig();
  }

  // ... get / set / reload / onChange / close 保持原有签名

  close(): void {
    this.stopWatching();
    this.listeners.clear();
  }

  // ========== 内部辅助 ==========

  private stopWatching(): void {
    if (this.debounceTimer) {
      clearTimeout(this.debounceTimer);
      this.debounceTimer = undefined;
    }
    if (this.watcher) {
      this.watcher.close();
      this.watcher = undefined;
    }
  }

  private watchConfig(): void {
    this.stopWatching();

    const onEvent = (eventType: string, filename: string | null): void => {
      if (filename !== null && filename !== basename(this.configPath)) return;
      this.debouncedReload();
    };

    if (existsSync(this.configPath)) {
      this.watcher = watch(this.configPath, (eventType, filename) => {
        onEvent(eventType, filename);
        // 文件可能被删除/重命名，需要重新建立监听
        if (!existsSync(this.configPath)) {
          this.watchConfig();
        }
      });
    } else {
      const dir = dirname(this.configPath);
      if (existsSync(dir)) {
        this.watcher = watch(dir, (eventType, filename) => {
          onEvent(eventType, filename);
          // 文件已创建，切换到文件监听
          if (existsSync(this.configPath)) {
            this.watchConfig();
          }
        });
      }
    }
  }

  private debouncedReload(): void {
    if (this.debounceTimer) clearTimeout(this.debounceTimer);
    this.debounceTimer = setTimeout(() => {
      this.reload();
    }, this.debounceMs);
  }
}
```

注意：`load()` 与 `reload()` 的现有实现保留；`startWatching()` 改名为 `watchConfig()`。
